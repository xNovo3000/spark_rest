void registerEndpoint(
    final String uri, final Method method, final Endpoint endpoint) {
  _router
      .putIfAbsent(
          uri, () => UriChain(middlewares: [], methodRouter: MethodRouter()))
      .methodRouter
      .putIfAbsent(
          method,
          () => EndpointChain(
              requestMiddlewares: [],
              endpoint: endpoint,
              responseMiddlewares: []));
  _context.register(endpoint);
}
